function updateAssembly(step) { if (stepDisplay) stepDisplay.innerText = step; const nodes = getNodes(); if (nodes.length === 0) return; nodes.forEach(node => { const name = node.name || ""; if (name.toLowerCase().includes("part_")) { const matches = name.match(/\d+/); if (matches) { const partNum = parseInt(matches[0]); // THE FIX: Change <= to === // We only want to see the objects that belong EXACTLY to this step const isVisible = (partNum === step); // Apply visibility to the node node.visible = isVisible; // Deep toggle children (the 2 meshes inside your branches) if (node.children) { node.children.forEach(child => { child.visible = isVisible; }); } // Force scale to 0 for hidden steps so they don't interfere with lighting if (node.scale) { const s = isVisible ? 1 : 0; node.scale.set(s, s, s); } } } }); }
