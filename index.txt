            <input type="range" min="1" max="5" value="5" class="slider" id="assemblySlider">
        </div>
    </model-viewer>

    <script>
        const viewer = document.getElementById('viewer');
        const slider = document.getElementById('assemblySlider');
        const stepDisplay = document.getElementById('stepDisplay');

        function updateAssembly(step) {
            stepDisplay.innerText = step;
            if (!viewer.model) return;

            for (let i = 1; i <= 5; i++) {
                // Must match the object names in Rhino (Part_001, etc.)
                const nodeName = "Part_" + i.toString().padStart(3, '0');
                const node = viewer.model.nodes.find(n => n.name === nodeName);
                if (node) {
                    node.visible = (i <= step);
                }
            }
        }

        slider.addEventListener('input', (e) => {
            updateAssembly(parseInt(e.target.value));
        });

        viewer.addEventListener('load', () => {
            updateAssembly(5);
        });
    </script>
